бсвр = ev
бсзр = pv
фсвр = ac

опс(cv) = бсвр - фсвр = (ev - ac)
окп(sv) = бсвр - бсзр = (ev - pv)

ИОС(cpi) = (ev/ac)
ИОКП(spi) = (ev/pv)

-------------------------------------------------------------
                    SV  < 0                    SV  > 0
                    SPI < 1                    SPI > 1
-------------------------------------------------------------
      CV  > 0      Экнонимя буджета           Экономия буджета
      CPI > 1      Отстование по срокам       Опережение сроков
-------------------------------------------------------------
      CV  < 0     Перерасход бюджета         Перерасход бюджета
      CPI < 1     Отстование по срокам       Опережение сроков
-------------------------------------------------------------
